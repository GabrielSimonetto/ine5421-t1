PROGRAM : FUNCTION

FUNCTION: DEF FUNC_IDENT OPENPAR CLOSEPAR STATEMENTLIST

STATEMENTLIST: STATEMENT
                | STATEMENTGENERATOR

STATEMENTGENERATOR: STATEMENT STATEMENTLIST

STATEMENT: DEFINEVARIABLE
            | ATTRIBUTION
            | CONDITIONAL
            | PRINT


DEFINEVARIABLE: TYPEDEFINITION VARIDENT CLOSE_STATEMENT_TOKEN

ATTRIBUTION: VARIDENT = VALUE CLOSE_STATEMENT_TOKEN

CONDITIONAL: if VARIDENT CONDOPS OPENBRACKET STATEMENTLIST CLOSEBRACKET

PRINT: print STRING CLOSE_STATEMENT_TOKEN
        | print VARIDENT CLOSE_STATEMENT_TOKEN

DEF: def
FUNC_IDENT: myFunc
OPENPAR: (
CLOSEPAR: )
OPENBRACKET: {
CLOSEBRACKET: }

TYPEDEFINITION: int
VARIDENT: i
VALUE: 0 | 4
CONDOPS: >
CLOSE_STATEMENT_TOKEN: ;
STRING: "i eh positivo"
